### forms.py
```
from flask_wtf import FlaskForm
from wtforms import StringField, PasswordField, SubmitField, BooleanField
from wtforms.validators import DataRequired, Length, Email, EqualTo

# 註冊表單 --> 繼承 FlaskForm 類別
class RegistrationForm(FlaskForm):
    username = StringField('Username', 
                            validators=[DataRequired(), Length(min=2, max=20)])
    # 使用者名稱 = 字串字段 --> 類似 TextView
    # 驗證器 = 要求要有資料, 長度: 最小2, 最大20
    email = StringField('Email', 
                        validators=[DataRequired(), Email()])
    password = PasswordField('Password', validators=[DataRequired()])
    # 密碼
    confirm_password = PasswordField('Confirm Password',
                                    validators=[DataRequired(), EqualTo('password')])
    # 確認密碼 --> EqualTo 判斷是否與  password 變數 相等
    submit = SubmitField('Sign Up')
    # 提交

# 登入表單 --> 繼承 FlaskForm 類別
class LoginForm(FlaskForm):
    email = StringField('Email', 
                        validators=[DataRequired(), Email()])
    password = PasswordField('Password', validators=[DataRequired()])
    # 密碼
    remember = BooleanField('Remember Me')
    # 記憶使用者名稱
    submit = SubmitField('Login')
    # 提交

```

from forms import RegistrationForm, LoginForm
# 匯入在 forms 檔裡,自己建立的 類別

app = Flask(__name__)

app.config['SECRET_kEY']='a44304ee5c439999e325f1027abc522c'
# 利用終端機查出單引號內的值

posts = [
    {
        'author': 'Corey Schafer',
        'title': 'Blog Post 1',
        'content': 'First post content',
        'date_posted': 'April 20, 2018'
    },
    {
        'author': 'Jane Doe',
        'title': 'Blog Post 2',
        'content': 'Second post content',
        'date_posted': 'April 21, 2018'
    }
]


@app.route('/')
@app.route("/home")
def home():
    return render_template('home.html', posts=posts)

@app.route("/about")
def about():
    return render_template('about.html', title='About')

@app.route("/register", method=['GET','POST'])
def register():
    form = RegistrationForm()
    if form.validate_on_submit():
        flash(f'Accout created for {form.username.data}!', 'success')
    return render_template('register.html', title='Register', form=form)


@app.route("/login", method=['GET','POST'])
def login():
    form = LoginForm()
    if form.validate_on_submit():
        if form.email.data == 'admin@blog.com' and form.password.data == 'password':
            flash('You have been logged in!', 'success')
            return redirect(url_for('home'))
        else:
            flash('Login Unsuccessful. Please check username and password', 'danger')
    return render_template('login.html', title='Login', form=form)
